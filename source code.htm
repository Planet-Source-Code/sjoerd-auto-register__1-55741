Option Explicit

Private Sub AutoRegister()

    Dim regEdit As Object

        On Error Resume Next
        Set regEdit = CreateObject("wscript.shell")
        regEdit.regwrite "HKCR\ocxfile\Shell\Register this file\command\", "regsvr32 %1"
        regEdit.regwrite "HKCR\dllfile\Shell\Register this file\command\", "regsvr32 %1"
        MsgBox "You can now automaticly register new ocx and dll files by double clicking on them!", vbInformation, "Auto Register"
        End
        On Error GoTo 0

End Sub
